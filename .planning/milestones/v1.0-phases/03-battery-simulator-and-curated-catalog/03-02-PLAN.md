<locked_physics>
<!-- RESEARCH.md Pattern 2 / §Simulation Math — Convention A (recommended). socKwh = energy IN the cell -->
<!-- (post-charge-loss). eff = Math.sqrt(roundTripEfficiency). usable = nominalCapacityKwh * dodFraction. -->
<!-- soc starts at 0 (D-06). Per interval, net = gridExportKwh - gridImportKwh (D-07): -->
<!--   net > 0 (charge): headroomGridSide = (usable - soc)/eff; -->
<!--                     gridSideCharge = min(net, maxChargeKw*h, headroomGridSide); -->
<!--                     soc = min(soc + gridSideCharge*eff, usable); -->
<!--                     chargedKwh = gridSideCharge (GRID-SIDE, A-1); residualExport = export - gridSideCharge; residualImport = 0. -->
<!--   net < 0 (discharge): demand = -net; delivered = min(demand, soc*eff, maxDischargeKw*h); -->
<!--                        soc -= delivered/eff (clamp soc >= 0); dischargedKwh = delivered; -->
<!--                        residualImport = demand - delivered; residualExport = 0. -->
<!-- shiftedKwh = sum of dischargedKwh. soc cap is HARD at usable (SIM-04). -->

<!-- CRITERION 2 (power clamp): 1.5 kWh export, 15-min (0.25h), Sessy 2.2 kW -> -->
<!--   gridSideCharge = min(1.5, 2.2*0.25=0.55, headroom) = 0.55; residualExport = 0.95. -->
<!-- CRITERION 3 (capacity-clamped, NOT pure round-trip): 5 kWh nominal @ 0.90 DoD = 4.5 usable. -->
<!--   Charge MORE than 4.5 so the cap engages (soc pins to 4.5), then discharge enough demand: -->
<!--   delivered = 4.5 * sqrt(0.9) = 4.269. A literal 6-in/6-out with NO cap gives 5.4 and FAILS. -->
<!--   Keep rte = 0.90 and sqrt each way; do NOT tweak rte to force the number. -->
</locked_physics>
